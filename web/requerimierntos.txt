from django.db import models
from django.core.exceptions import ValidationError


class Usuario(models.Model):
    id = models.UUIDField(primary_key=True, editable=False)
    folio = models.CharField(max_length=10, unique=True)
    email = models.EmailField(unique=True)
    tipo = models.ForeignKey('TipoUsuario', on_delete=models.PROTECT)

    propietario = models.OneToOneField('Propietario', null=True, blank=True, on_delete=models.CASCADE)
    recepcionista = models.OneToOneField('Recepcionista', null=True, blank=True, on_delete=models.CASCADE)
    veterinario = models.OneToOneField('Veterinario', null=True, blank=True, on_delete=models.CASCADE)
    administrador = models.OneToOneField('Administrador', null=True, blank=True, on_delete=models.CASCADE)

    imagen = models.ImageField(upload_to='usuarios/', null=True, blank=True)

    def _str_(self):
        return self.folio

    def clean(self):
        roles = {
            'PROP': self.propietario,
            'REC': self.recepcionista,
            'VET': self.veterinario,
            'ADMIN': self.administrador
        }

        roles_llenos = [r for r in roles.values() if r is not None]

        if len(roles_llenos) != 1:
            raise ValidationError("El usuario debe tener exactamente un rol asignado")

        if self.tipo.codigo == 'PROP' and not self.propietario:
            raise ValidationError("El tipo es PROPIETARIO pero no se asignó propietario")

        if self.tipo.codigo == 'REC' and not self.recepcionista:
            raise ValidationError("El tipo es RECEPCIONISTA pero no se asignó recepcionista")

        if self.tipo.codigo == 'VET' and not self.veterinario:
            raise ValidationError("El tipo es VETERINARIO pero no se asignó veterinario")

        if self.tipo.codigo == 'ADMIN' and not self.administrador:
            raise ValidationError("El tipo es ADMIN pero no se asignó administrador")

    def save(self, *args, **kwargs):
        self.clean()
        super().save(*args, **kwargs)
